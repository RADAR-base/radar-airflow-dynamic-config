@dag(dag_id="train_model-v2", schedule="@daily",
     start_date=datetime(2025, 1, 1),
     tags=["airflow101"], catchup=False, default_args=default_args)
def train_wine_quality():
    """
    This DAG trains a linear regression model using the ElasticNet algorithm.
    It uses the wine quality dataset and evaluates the model's performance using RMSE, MAE, and R2 metrics.
    """

    @task.bash
    def conda_env() -> str:
        return 'source /Users/heetsankesara/miniconda3/etc/profile.d/conda.sh && conda activate wine-quality'

    @task.bash
    def create_temp_directory() -> str:
        return 'mkdir -p ./tmp'

    @task.bash
    def remove_temp_directory() -> str:
        return 'rm -rf ./tmp'

    @task(do_xcom_push=True, multiple_outputs=True)
    def read_data_task(args):
        # Save read data
        train_x, test_x, train_y, test_y = read_data(args)
        # Save the data to local storage
        xcom_dict = {
            "train_x": train_x,
            "test_x": test_x,
            "train_y": train_y,
            "test_y": test_y
        }
        return xcom_dict

    @task()
    def train_model_task(args, return_data_xcom):
        # Unpack the data from xcom
        train_x = return_data_xcom["train_x"]
        test_x = return_data_xcom["test_x"]
        train_y = return_data_xcom["train_y"]
        test_y = return_data_xcom["test_y"]
        lr = train_model(args, train_x, test_x, train_y, test_y)
        # Save the model to xcom
        # Save the model to local storage
        pickle_path = save_model(lr)
        return {"pickle_path": pickle_path}

    @task(do_xcom_push=True)
    def eval_model_task(args, return_data_xcom, lr_xcom):
        # read model
        pickle_path = lr_xcom['pickle_path']
        # read the pickle file
        with open(pickle_path, "rb") as f:
            lr = pickle.load(f)

        # Unpack the data from xcom
        test_x = return_data_xcom["test_x"]
        test_y = return_data_xcom["test_y"]
        # Unpack the model from xcom
        rmse, mae, r2 = eval_model(args, test_x, test_y, lr)
        return_xcom_metrics = {
            "rmse": rmse,
            "mae": mae,
            "r2": r2
        }
        return return_xcom_metrics

    @task
    def save_metrics_data(metric_data_xcom):
        # Unpack the metrics from xcom
        rmse = metric_data_xcom["rmse"]
        mae = metric_data_xcom["mae"]
        r2 = metric_data_xcom["r2"]
        # Save the metrics to a file
        metrics = {
            "rmse": rmse,
            "mae": mae,
            "r2": r2
        }
        metrics_df = pd.DataFrame(metrics, index=[0])
        metrics_df.to_csv("metrics.csv", index=False)

    @task
    def print_something_task():
        print("Hello world")

    # convert dict to args
    args = argparse.Namespace(**default_args)
    #task_1 = create_temp_directory().set_upstream(print_something_task())
    return_data_xcom = read_data_task(args)

    #train_x, test_x, train_y, test_y = read_data(args)
    ## Train the model
    #
    lr_xcom = train_model_task(args, return_data_xcom)
    ## Evaluate the model
    metrics_df_xcom = eval_model_task(args, return_data_xcom, lr_xcom)
    save_metrics_data(metrics_df_xcom)
